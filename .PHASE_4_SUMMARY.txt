📊 STATE MANAGEMENT (Zustand)
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

useTaskStore:

State:
  ├─ currentTask: StoredTask | null
  ├─ tasks: Record<string, StoredTask>
  ├─ events: AgentEvent[]
  ├─ isLoading: boolean
  └─ error: string | null

Actions:
  ├─ createTask(request, taskId)
  ├─ addEvent(event)
  ├─ setCurrentTask(taskId)
  ├─ setTaskStatus(taskId, status)
  ├─ setTaskResult(taskId, result)
  ├─ clearError()
  ├─ setError(error)
  ├─ setLoading(loading)
  └─ reset()

🔄 FLUJO DE DATOS COMPLETO
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

Usuario en Browser:
  1. Abre http://localhost:3000
  2. Dashboard renderiza 3 áreas
  3. WebSocket se conecta automáticamente
     │
  4. Completa formulario en TaskInput
     │
  5. Click "Enviar Tarea"
     │ (POST /api/task)
     ├─→ Gateway (Go)
         ├─→ AI Worker (Python)
         │   ├─ Manager analiza
         │   ├─ Especialista ejecuta
         │   └─ QA revisa
         └─ Publica eventos a Redis
              │
  6. Redis channel: "agency_events"
     │
  7. Gateway WebSocket escucha
     │
  8. Broadcast a clientes conectados
     │
  9. Frontend WebSocket recibe
     │ (useWebSocket hook)
     ├─→ Zustand store addEvent()
     │
  10. LiveFeed renderiza eventos
      ├─ Avatar + nombre agente
      ├─ Acción + mensaje
      └─ Timestamp
     │
  11. Output panel muestra resultado
      cuando se completa
